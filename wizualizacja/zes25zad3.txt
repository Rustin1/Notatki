#include <stdio.h>
#include <stdlib.h>

void foo(int ** tab, int n, int m)
{
	for(int i=0,i<n,i++)
	{
		tab[i][0]= tab[i][0]*tab[i][0];
		tab[i][m-1]= tab[i][m-1]*tab[i][m-1];
	}
}

void wypisz(int ** tab, int n, int m)
{
	for(int i=0,i<n,i++)
	{
		for(int j=0;j<m;j++)
		{
			printf("[%d,%d]=%d, ",i,j,tab[i][j]);
		}
	}
}

int main()
{
	tab[0]= malloc(3*sizeof(int));
	tab[1]= malloc(3*sizeof(int));
	tab[2]= malloc(3*sizeof(int));
	tab[3]= malloc(3*sizeof(int));
	tab[0][0]=1;
	tab[0][1]=2;
	tab[0][2]=3;
	tab[1][0]=0;
	tab[1][1]=5;
	tab[1][2]=8;
	tab[2][0]=9;
	tab[2][1]=8;
	tab[2][2]=-2;
	tab[3][0]=-2;
	tab[3][1]=3;
	tab[3][2]=4;
	wypisz(tab,4,3);
	foo(tab,4,3);
	wypisz(tab,4,3);
	return 0;
}





funkcja, której argumentem jest dwuwymiarowa tablica tablic(ma zmienne typu int)
oraz jej wymiary n i m (n>0,m>1). funkcja ma podnieść do kwadratu element z pierwszej
i ostatniej kolumny
